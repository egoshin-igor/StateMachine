public void Main( List<string> args)
{
	List<int> numbers;
	BubbleSort(numbers);
}

public void BubbleSort(List<int> numbersToSort)
{
	bool isSorted = false;
	while( !isSorted )
	{
		isSorted = true;
		for(int i = 1; i < numbersToSort.Length; i++)
		{
			int prevNumber;
			int currNumber;
			prevNumber  = numbersToSort[i - 1];
			currNumber = numbersToSort[i];
			if (prevNumber > currNumber)
			{
				Swap( numbersToSort, i, i - 1 );
				isSorted = false;
			}
		}
	}
}

public void Swap( List<int> numbers, int i1, int i2)
{
	int temp;
	temp = numbers[i1];
	numbers[i1] = numbers[i2];
	numbers[i2] = temp;
}
